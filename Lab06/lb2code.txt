<script type="text/javascript">
function average(a, b, c, d, e) {
    return (parseFloat(a) + parseFloat(b) + parseFloat(c) + parseFloat(d) + parseFloat(e)) / 5;
}


var a = prompt("Enter the value for 'a':");
var b = prompt("Enter the value for 'b':");
var c = prompt("Enter the value for 'c':");
var d = prompt("Enter the value for 'd':");
var e = prompt("Enter the value for 'e':");


var calculatedAverage = average(a, b, c, d, e);
if (isNaN(calculatedAverage)) {
    alert("Please enter valid numeric values.");
} else {
    if (calculatedAverage >= 30) {
        alert("Congratulations! You Passed.");
    } else {
        alert("Try again! You Failed.");
    }
}
